# APT v2 — W3 "Forge" Rebrand Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the Iris (indigo) Model-Hub brand with the approved **Forge** identity — warm-graphite + one ember-copper accent + dimensional die-mark logo — across web tokens, Tailwind, design docs, TUI themes, landing page, installer, README, and a new committed `assets/` directory, with zero iris hexes left anywhere outside git history.

**Architecture:** The web app is fully token-driven (CSS variables in `web/tokens.css`, mirrored in `web/src/index.css`, mapped to Tailwind utility names in `web/tailwind.config.ts`), so the re-skin is a token swap plus a mechanical `iris` → `copper` utility rename — **no layout or component changes**. Python-side color truth moves into a new `backend/brand.py` constants module that the TUI themes import (W2's TUI rework consumes the same constants). The die mark is committed as canonical SVGs; raster icons (.png/.ico/social card) are generated by a committed Pillow script and the outputs are committed too.

**Tech Stack:** CSS variables, Tailwind CSS 3 + shadcn (Vite + React + TS), Textual `Theme` API (Python), Pillow (dev-only, for icon rasterization), InnoSetup, PyInstaller, GitHub Actions.

```python
"""
Generate apt's raster brand assets from the die-mark geometry.

One-time generator; all outputs are COMMITTED. Re-run only when the mark
changes. Pillow is a dev-only tool (build.spec excludes PIL from the exe;
it is NOT a runtime dependency and must not enter requirements.txt).

Usage (from repo root):
    .venv\\Scripts\\python.exe -m pip install pillow
    .venv\\Scripts\\python.exe assets/generate_icons.py
"""
from __future__ import annotations

import shutil
from pathlib import Path

from PIL import Image, ImageDraw

ROOT = Path(__file__).resolve().parent.parent
ASSETS = ROOT / "assets"
ICONS = ASSETS / "icons"

# --- Die-mark geometry (design box 0..100) -----------------------------------
# MUST stay in sync with assets/die-mark.svg (the canonical vector).
TOP_FACE = [(50, 24), (84, 43), (50, 62), (16, 43)]
LEFT_FACE = [(16, 43), (50, 62), (50, 74), (16, 55)]
RIGHT_FACE = [(50, 62), (84, 43), (84, 55), (50, 74)]
TRACE = [(28, 44), (46, 34), (62, 43), (72, 38)]
DOT_CENTER = (72, 38)
DOT_RADIUS = 3.0
TRACE_WIDTH = 2.5
GLYPH_CENTER_Y = 49.0  # midpoint of the mark's y-extent (24..74)

TOP_FILL = (0x2E, 0x28, 0x25, 255)
LEFT_FILL = (0x19, 0x15, 0x12, 255)
RIGHT_FILL = (0x22, 0x1D, 0x1A, 255)
COPPER = (0xFF, 0x6A, 0x3D, 255)
PLATE = (0x0C, 0x0A, 0x09, 255)  # --bg, warm near-black

SS = 8  # supersampling factor: render big, downscale with LANCZOS


def _draw_mark(draw: ImageDraw.ImageDraw, ox: float, oy: float, scale: float) -> None:
    """Draw the die mark; design box top-left at (ox, oy), scaled by `scale`."""

    def pt(p):
        return (ox + p[0] * scale, oy + p[1] * scale)

    draw.polygon([pt(p) for p in TOP_FACE], fill=TOP_FILL)
    draw.polygon([pt(p) for p in LEFT_FACE], fill=LEFT_FILL)
    draw.polygon([pt(p) for p in RIGHT_FACE], fill=RIGHT_FILL)

    w = TRACE_WIDTH * scale
    points = [pt(p) for p in TRACE]
    draw.line(points, fill=COPPER, width=max(1, round(w)), joint="curve")
    for x, y in points:  # round caps + joints
        draw.ellipse([x - w / 2, y - w / 2, x + w / 2, y + w / 2], fill=COPPER)

    cx, cy = pt(DOT_CENTER)
    r = DOT_RADIUS * scale
    draw.ellipse([cx - r, cy - r, cx + r, cy + r], fill=COPPER)


def render_mark(size: int) -> Image.Image:
    """Transparent-background mark, glyph vertically centred on the canvas."""
    big = size * SS
    img = Image.new("RGBA", (big, big), (0, 0, 0, 0))
    scale = big / 100.0
    oy = (50.0 - GLYPH_CENTER_Y) * scale
    _draw_mark(ImageDraw.Draw(img), 0.0, oy, scale)
    return img.resize((size, size), Image.LANCZOS)


def render_app_icon(size: int) -> Image.Image:
    """Dark rounded plate + mark at 78% — Windows/app icon."""
    big = size * SS
    img = Image.new("RGBA", (big, big), (0, 0, 0, 0))
    d = ImageDraw.Draw(img)
    d.rounded_rectangle([0, 0, big - 1, big - 1], radius=big * 0.22, fill=PLATE)
    scale = big * 0.78 / 100.0
    ox = (big - 100.0 * scale) / 2
    oy = (big - 100.0 * scale) / 2 + (50.0 - GLYPH_CENTER_Y) * scale
    _draw_mark(d, ox, oy, scale)
    return img.resize((size, size), Image.LANCZOS)


def render_social_preview() -> Image.Image:
    """1280x640 GitHub social card: plate + big mark, no text.

    Text is deliberately omitted: GitHub overlays the repo name on the card,
    and rendering the wordmark would need a bundled font. Mark only.
    """
    w, h = 1280 * 2, 640 * 2
    img = Image.new("RGBA", (w, h), PLATE)
    d = ImageDraw.Draw(img)
    scale = (h * 0.62) / 100.0
    ox = (w - 100.0 * scale) / 2
    oy = (h - 100.0 * scale) / 2 + (50.0 - GLYPH_CENTER_Y) * scale
    _draw_mark(d, ox, oy, scale)
    return img.resize((1280, 640), Image.LANCZOS).convert("RGB")


def main() -> None:
    ICONS.mkdir(parents=True, exist_ok=True)

    for size in (16, 32, 48, 64, 128, 256, 512):
        render_mark(size).save(ICONS / f"die-mark-{size}.png")
        render_app_icon(size).save(ICONS / f"app-icon-{size}.png")

    base = render_app_icon(256)
    base.save(
        ASSETS / "app-icon.ico",
        format="ICO",
        sizes=[(16, 16), (24, 24), (32, 32), (48, 48), (64, 64), (128, 128), (256, 256)],
    )
    base.save(
        ASSETS / "favicon.ico",
        format="ICO",
        sizes=[(16, 16), (32, 32), (48, 48)],
    )

    render_social_preview().save(ASSETS / "social-preview.png")

    # Copies consumed by the web app (Vite serves web/public at /) and the site
    (ROOT / "web" / "public").mkdir(exist_ok=True)
    shutil.copyfile(ASSETS / "favicon.ico", ROOT / "web" / "public" / "favicon.ico")
    shutil.copyfile(ASSETS / "die-mark.svg", ROOT / "web" / "public" / "favicon.svg")
    shutil.copyfile(ASSETS / "die-mark.svg", ROOT / "site" / "favicon.svg")

    print(f"brand assets written under {ASSETS}")


if __name__ == "__main__":
    main()
```
